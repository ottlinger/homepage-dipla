007 import java.sql.Connection;
008 import java.sql.PreparedStatement;
009 import java.sql.ResultSet;
010 import java.sql.SQLException;
011 
012 import javax.sql.DataSource;
013 
014 import org.apache.log4j.Logger;
015 
016 import de.fub.tip.datenanzeige.container.InformationVOContainer;
017 import de.fub.tip.datenanzeige.ormapper.InformationVO;
018 import de.fub.tip.datenanzeige.ormapper.PictureVO;
019 import de.fub.tip.datenanzeige.ormapper.SightVO;
020 import de.fub.tip.datenanzeige.ormapper.TopicVO;
021 import de.fub.tip.datenanzeige.ormapper.UserdataVO;
022 import de.fub.tip.datenbank.factory.LogicObject;
023 import de.fub.tip.exceptions.NoDataFoundException;
024 import de.fub.tip.exceptions.NoUserLoggedInException;

027 /**
028  * Sinn: zeigt Informationen zu Themen an
029  * @author hirsch, 10.06.2004
030  * @version 2004-06-10
031  * $Id: InformationsAnzeiger.java,v 1.17 2004/06/11 19:36:46 hirsch Exp $
032  */
033 public class InformationsAnzeiger implements LogicObject {
034   /** Logger zur Fehlersuche */
035   private static Logger logger = 
036       Logger.getLogger(InformationsAnzeiger .class);
037   /** Datenquelle mit der der <code>InformationsAnzeiger</code> arbeitet */
038   private DataSource datasource = null;
039 
040   /**
041    * Standardkonstruktor, der einen InformationsAnzeiger Bearbeiter erzeugt  
042    * @param datasource Datenquelle
043    */
044   public InformationsAnzeiger (DataSource datasource) {
045     this.datasource = datasource;
046     logger.info("InformationsAnzeiger initialized.");
047   } // end of Konstruktor
048   
049   /**
050    * zeigt alle Informationen zu einer Sehenswürdigkeit nacheinander an!
051    * 
052    * @param sight - Sehenswürdigkeit, zu der Infos angezeigt 
053    * werden sollen
054    * @param user - angemeldeter Benutzer
055    * 
056    * @return <i>Container mit InformationVO</i>-Objekten
057    * 
058    * @throws NoDataFoundException, wenn keine Daten gefunden wurden
059    *   oder der Benutzer eine falsche URL übergeben hat.
060    * 
061    * @throws SQLException, wenn Datenbankprobleme auftreten
062    */
063   public InformationVOContainer getAllInformationForSight(SightVO sight, 
064           UserdataVO user) 
065     throws NoDataFoundException, SQLException, NoUserLoggedInException {
066       
067       if ( user == null) throw new NoUserLoggedInException(
068               "InformationsAnzeiger.getAllInfoForSight - unbekannter Benutzer");
069       
070     String sql = "SELECT DISTINCT s.*, s.name AS sname, i.*, t.*, t.name AS tname, p.*  "+
071         "FROM sight s, information i, userdata u, profile pr,  " +
072         "topicprofile topr, topic t, picture p  "+
073         "WHERE s.si_id=i.id AND u.udid = pr.udid " +
074         "AND topr.pid = pr.udid AND topr.tid = t.tid " +
075         "AND p.id = s.picture_url " +
076         "AND pr.udid= ? AND topr.tid = ? AND s.si_id= ?" +
077         "AND t.tid IN (SELECT tt.tid from topicprofile tt WHERE tt.pid= ?) ";
078 
079 
080     logger.debug("getAllInfo4Sights() - SightID = "+ sight.getId());
081     Connection con = this.datasource.getConnection();
082     // Parameter setzen
083     PreparedStatement prepStatement = con.prepareStatement(sql);
084     prepStatement.setString(1, user.getId().toString());
085     prepStatement.setString(2, sight.getId().toString());
086     prepStatement.setString(3, sight.getId().toString());
087     prepStatement.setString(4, user.getId().toString());
088     
089     boolean success = prepStatement.execute();
090     if ( success ) {
091       logger.debug(
092       "InformationsAnzeiger.getAllInfos4Sight(): " +
093       "PreparedStatement erfolgreich ausgeführt.");
094     } else {
095       logger.debug(
096           "InformationsAnzeiger.getAllInfos4Sight(): " +
097         "PreparedStatement-Fehler!!! ");
098       throw new NoDataFoundException(
099         "InformationsAnzeiger.getAllInfos4Sight()");
100     } // end if
101   
102     // Ergebnis in Beans umwandeln
103       InformationVOContainer ergebnis = new InformationVOContainer();
104     InformationVO info = null;
105     TopicVO topic = null;
106     SightVO ssight = null;
107     PictureVO picture = null;
108     URL url = null;
109     
110     ResultSet rs = prepStatement.getResultSet();
111     if( rs != null) {
112       while(rs.next()) {
113         // Information VO zusammenbasteln 
114         info = new InformationVO();
115         topic = new TopicVO();
116         ssight = new SightVO();
117         picture = new PictureVO();
118         
119         try {
120             // aboStatus default auf falsch, da das hier egal ist!
121             topic.setAboStatus(new Boolean(false));
122             topic.setDescription(rs.getString("description"));
123             topic.setName(rs.getString("tname"));
124 logger.debug("topic = "+ rs.getString("tname") + rs.getString("description"));
125 
126           // Bildinfos zusammenbasteln
127           picture.setHeight(rs.getString("height"));
128           picture.setHeightThumb(rs.getString("heightthumb"));
129           picture.setId(new Integer(rs.getInt("picture_url")));
130 
131           url = new URL(rs.getString("url"));
132           picture.setUrl(url);
133           url = new URL(rs.getString("urlthumb"));
134           picture.setUrlThumb(url);
135 
136           picture.setWidth(rs.getString("width"));
137           picture.setWidthThumb(rs.getString("widththumb"));
138           
139             ssight.setId(new Integer(rs.getInt("si_id")));
140             ssight.setName(rs.getString("sname"));
141             ssight.setLocation(rs.getString("location"));
142             ssight.setPicture(picture);
143 logger.debug("ssight = " + rs.getString("location")+ rs.getString("sname"));
144             
145             info.setId(new Integer(rs.getInt("iid")));
146           info.setInformation(rs.getString("information"));
147           info.setLevel(new Integer(rs.getString("lvl")));
148           info.setSight(ssight);
149           info.setTopic(topic);
150 
151           logger.debug("HIER! Info: "+info.getId()+ "/"+info.getInformation() + "sight - "+info.getSight().getId());
152         } catch(Exception e) {
153           // Probleme gibt es hier beispielsweise bei NULL-Werten!
154           // Diese werden einfach gefangen und generieren keinen Fehler
155           logger.error(
156               "InformationsAnzeiger.getAllInfos4Sight(): "+
157             e + " - Ausnahme beim Parsen der Daten! " +
158             e.getLocalizedMessage());
159         } // end of catch
160   
161         // zur Ergebnismenge hinzufügen und probeweise loggen
162         ergebnis.add(info);
163       } // end of while
164     } // end if
165   
166     // Schliessen/Freigabe der Datenbankverbindung
167     try {
168       if( con != null ) con.close();
169       logger.debug(
170           "InformationsAnzeiger.getAllInfos4Sight(): " +
171         "DB-Con geschlossen.");
172     } catch (SQLException e) {
173       logger.debug(
174           "InformationsAnzeiger.getAllInfos4Sight():" +
175         "geschachtelte Ausnahme beim Schliessen der " +
176         "DB-Verbindung.");
177     } // end of try
178   
179     // nix da ?      
180     if(ergebnis.size() <= 0) {
181       logger.debug(
182            "InformationsAnzeiger.getAllInfos4Sight(): " +
183         "keine Sehenswürdigkeiten da"); 
184       throw new NoDataFoundException(
185               "InformationsAnzeiger.getAllInfos4Sight(): " +
186           "keine Daten gefunden");
187     } // end if
188   
189     logger.debug("InformationsAnzeiger.getAllInfos4Sight(): " +
190       "hat " + ergebnis.size() + " Elems.");
191   
192     return ergebnis;
193   } // end of getAllInformationForSight
194 } // end of class

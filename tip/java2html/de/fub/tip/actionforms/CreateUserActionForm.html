013 /**
014  * Sinn: stellt die HTML-Seite zur Verfügung, mit der ein neuer Benutzer angelegt 
015  * werden kann.<br>
016  * Inhaltlich handelt es sich nur um Eingabefelder für:
017  * <ul>
018  * <li>Benutzername</li>
019  * <li>Passwort</li>
020  * <li>Passwort-Wiederholung</li>
021  * <li>Vorname</li>
022  * <li>Nachname</li>
023  * </ul>
024  * <br>
025  * 
026  * @author hirsch, 17.04.2004
027  * @version 2004-04-25
028  * $Id: CreateUserActionForm.java,v 1.6 2004/04/26 08:03:17 hirsch Exp $
029  * 
030  */
031 public class CreateUserActionForm extends ActionForm {
032   /** HTML: Vorname des Benutzers-Feld */
033   private String name = null;
034   /** HTML: Nachname des Benutzers-Feld */
035   private String lastname = null;
036   /** HTML: benutzername-Feld */
037   private String username = null;
038   /** HTML: passwort-Feld */
039   private String password = null;
040   /** HTML: passwortWiederholungs-Feld */
041   private String passwordAgain = null;
042 
043   /** alle gesammelten Fehler dieser Form */
044   private ActionErrors errors = null;
045   /** der aktuell bearbeitete Fehler */
046   private ActionError error = null;
047      
048 
049   /**
050    * wenn man im HTML-Formular auf "reset" drückt, werden alle Felder
051    * der Seite gelöscht.
052    * 
053    * @see org.apache.struts.action.ActionForm#reset(org.apache.struts.action.ActionMapping, javax.servlet.http.HttpServletRequest)
054    */
055   public void reset(ActionMapping am, HttpServletRequest hsr) {
056     this.password = "";
057     this.passwordAgain = "";
058     this.lastname = "";
059     this.name = "";
060     this.username = "";
061   } // end of reset
062 
063   /**
064    * diese Methode findet nur syntaktische Fehler - die
065    * semantische Prüfung sollte <b>nur</b> auf Geschäftslogikebene
066    * also in den Beans stattfinden!
067    * (sonst nicht mehr MVC)
068    *
069    * @see org.apache.struts.action.ActionForm#validate(org.apache.struts.action.ActionMapping, javax.servlet.http.HttpServletRequest)
070    * @param actionMapping
071    * @param request
072    * @return ActionErrors aufgetretene Fehler
073    * */
074   public ActionErrors validate(ActionMapping actionMapping,
075       HttpServletRequest request) {
076       errors = new ActionErrors();
077       
078     // überhaupt angegeben ?
079      if ( username == null || username.length() < 1 ) {
080            error = new ActionError("errors.username.required");
081            errors.add("username", error);
082      } // end if username
083      
084      if ( password == null || password.length() < 1 ) {
085       error = new ActionError("errors.password.required");
086       errors.add("password", error);
087        } // end if password
088 
089      if ( passwordAgain == null || passwordAgain.length() < 1 ) {
090            error = new ActionError("errors.password.required");
091            errors.add("passwordAgain", error);
092        } // end if passwordAgain
093      
094      // sind die beiden Passwörter gleich
095      if ( password.compareTo(passwordAgain) != 0 ) {
096          error = new ActionError("errors.passwordmismatch");
097          errors.add("password", error);
098          errors.add("passwordAgain", error);
099      } // end if
100 
101      // Vorname
102      if ( name == null || name.length() < 1 ) {
103            error = new ActionError("errors.firstnamerequired");
104            errors.add("name", error);
105        } // end if name
106      
107      // Nachname
108      if ( lastname == null || lastname.length() < 1 ) {
109            error = new ActionError("errors.lastnamerequired");
110            errors.add("lastname", error);
111        } // end if lastname
112      
113      return errors;
114    } // end of validate
115   
116    /**
117   * aktuelles Passwort zurückgeben
118   * @return gesetztes Passwort
119   */
120    public String getPassword() {
121      return password;
122    }
123 
124    /**
125   * aktuelles Passwort setzen
126   * @param password neues Passwort

128    public void setPassword(String password) {

130    }

135   */
136    public String getUsername() {

138    }
139

141   * Benutzernamen setzen
142   * @param username neuer Benutzername
143   */
144    public void setUsername(String username) {
145      this.username = username;
146    }
147    
148   /**
149    * gibt den Nachnamen des Benutzers zurück
150    * @return Nachname
151    */
152   public String getLastname() {
153     return lastname;
154   }
155 
156   /**
157    * setzt den Nachnamen des Benutzers neu
158    * @param lastname neuer Nachname
159    */
160   public void setLastname(String lastname) {
161     this.lastname = lastname;
162   }
163 
164   /**
165    * gibt den Vornamen des Benutzers zurück
166    * @return vorname
167    */
168   public String getName() {
169     return name;
170   }
171 
172   /**
173    * setzten den Vornamen des Benutzers neu
174    * @param name neuer Vorname
175    */
176   public void setName(String name) {
177     this.name = name;
178   }
179   
180     /**
181      * gibt den Wert des wiederholten Passworts zurück
182      * @return Returns the passwordAgain.
183      */
184     public String getPasswordAgain() {
185         return this.passwordAgain;
186     }
187     
188     /**
189      * setzt das wiederholte Passwort auf einen neuen Wert
190      * @param passwordAgain The passwordAgain to set.
191      */
192     public void setPasswordAgain(String passwordAgain) {
193         this.passwordAgain = passwordAgain;
194     }
195 } // end of class

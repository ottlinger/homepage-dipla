010 /**
011  * Sinn: kapselt den Inhalt der Tabelle information aus der Datenbank
012  * 
013  * @author hirsch, 30.03.2004

016  * 
017  */
018 public class InformationVO implements ViewObject, Serializable {
019   /** datenbankinterne ObjektID */
020   private Integer id = null;
021   /** Sehenswürdigkeit, der die Information zugeordnet ist */
022   private SightVO sight = null;
023   /** Thema, zu dem diese Information gehört */
024   private TopicVO topic = null;
025   /** interner Grad, den diese Information hat.<br>
026    * Diese Information dient der Speicherung des Zustands des hierarchischen
027    * Datenmodells.<br>
028    */
029   private Integer level = null;
030   /** der textuelle Inhalt dieser Information */ 
031   private String information = null;
032   
033   /**
034    * leerer Konstruktor, die eigentlichen Daten werden anders gefüllt - <br>
035    * nämlich direkt mit den Settern! 
036    * <br>
037    */
038   public InformationVO() {
039   } // end of Konstruktor
040 
041   /** 
042    * @see de.fub.tip.datenanzeige.ormapper.ViewObject#logObject(org.apache.log4j.Logger)
043    */
044   public void logObject(Logger logger) {
045     logger.debug("InformationVO.logObject("+ this.toString() +")");
046   } // end of logObject
047   
048   /**
049    * gibt Information als nullpointersicheren String zurück
050    * @return infoAlsString
051    */
052   public String toString() {
053     String ergebnis ="(";
054     
055     if ( this.getId() != null ) ergebnis += this.getId();
056     ergebnis += ") ";
057     if ( this.getInformation() != null ) ergebnis += this.getInformation();
058     ergebnis += ", Level=";
059     if ( this.getLevel() != null ) ergebnis += this.getLevel(); 
060     ergebnis += ", ";
061     if ( this.getSight() != null ) ergebnis += this.getSight();
062     ergebnis += ", ";
063     if ( this.getTopic() != null ) ergebnis += this.getTopic();
064     
065     return ergebnis;
066   } // end of toString
067 
068   /**
069    * gibt die DatenbankObjektID zurück
070    * @return ID
071    */
072   public Integer getId() {
073     return id;
074   }
075 
076   /**
077    * setzt die ObjektID

082   }

090   }
091

103    */
104   public Integer getLevel() {

106   }

120   public SightVO getSight() {
121     return sight;
122   }
123

125    * setzt die Sehenswürdigkeit

128   public void setSight(SightVO sight) {

130   }

136   public TopicVO getTopic() {

144   public void setTopic(TopicVO topic) {
145     this.topic = topic;

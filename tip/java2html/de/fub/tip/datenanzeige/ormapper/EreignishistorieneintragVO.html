020 public class EreignishistorieneintragVO implements ViewObject, Serializable {
021   /** datenbankinterne ID des Objekts */
022   private Integer id = null;
023   /** der zugehörige Benutzer */
024   private UserdataVO user = null;
025   /** die zugehörige/ausgelieferte Information */
026   private InformationVO information = null;
027   /** die aktuelle Sehenswürdigkeit */
028   private SightVO sight = null;
029 
030   /** der Ort der Sehenswürdigkeit <br> Da vermutlich auch 
031    * mehr als Punkte gespeichert werden könne, existiert hier kein
032    * eigenes ViewObjekt. Stattdessen wird die <code>toString()</code>-
033    * Methode des jeweiligen geometrischen Objeketes benutzt.<br>
034    * Später kann man da ja mal einen Wrapper schreiben, der das ganze nett
035    * verpackt! 
036    */
037   private PGgeometry location = null; 
038   /** der Zeitpunkt des Ereignisses*/
039   private Date time = null;
040 
041   /**
042    * leerer Konstruktor, die eigentlichen Daten werden anders gefüllt - <br>
043    * nämlich direkt mit den Settern! 
044    * <br>
045    */
046   public EreignishistorieneintragVO() {
047   } // end of Konstruktor
048 
049   /** 
050    * @see de.fub.tip.datenanzeige.ormapper.ViewObject#logObject(org.apache.log4j.Logger)
051    */
052   public void logObject(Logger logger) {
053     logger.debug("EreignishistorieVO.logObject(" + this.toString() + ")");
054   } // end of logObject
055   
056   /**
057    * gibt ein Ereignis als nullpointersicheren 
058    * String zurück
059    * @return ereignisString
060    */
061   public String toString() {
062     String ergebnis = "(";
063     if ( this.getId() != null ) ergebnis+= this.getId();
064     ergebnis += "), ";
065     if ( this.getUser() != null ) ergebnis+= this.getUser();
066     ergebnis += ", ";
067     if ( this.getInformation() != null ) ergebnis += this.getInformation();
068     ergebnis += "@";
069     if ( this.getLocation() != null ) ergebnis+= this.getLocation();
070     ergebnis += ", ";
071     if ( this.getSight() != null ) ergebnis+= this.getSight();
072     ergebnis += ", ";
073     if ( this.getTime() != null ) ergebnis += this.getTime();
074     
075     return ergebnis;
076   } // end of toString
077 
078   /**
